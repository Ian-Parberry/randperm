84template<class t> void Randomize(std::vector<t>& v){
85 std::default_random_engine stdRandom; //pseudo-random number generator
86 stdRandom.seed(GetEpochTime<unsigned>()); //sow seed
87
88 for(size_t i=0; i<v.size(); i++)
89 std::swap(v[i], v[std::uniform_int_distribution<size_t>(0, i)(stdRandom)]);
90} //Randomize
